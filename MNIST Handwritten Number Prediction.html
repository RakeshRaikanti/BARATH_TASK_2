# Make predictions on the test data
y_pred = model.predict(x_test)
y_pred_labels = np.argmax(y_pred, axis=1)

# Create a grid for visualization
num_rows = 6
num_cols = 6
num_images = num_rows * num_cols

# Select a random subset of test data for visualization
random_indices = np.random.choice(len(x_test), num_images, replace=False)

# Create a grid of subplots
fig, axes = plt.subplots(num_rows, num_cols, figsize=(12, 12))

for i, ax in enumerate(axes.flat):
    # Get the index of the random test example
    index = random_indices[i]
    
    # Display the image
    ax.imshow(x_test[index].reshape(28, 28), cmap='gray')
    
    # Set the actual and predicted labels as the title
    actual_label = y_test[index]
    predicted_label = y_pred_labels[index]
    ax.set_title(f"Actual: {actual_label}, Predicted: {predicted_label}")
    
    ax.axis('off')

# Adjust the layout
plt.tight_layout()
plt.show()
